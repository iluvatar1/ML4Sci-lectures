# defino la funcion para corregir los nombres:
def corregir_nombres(w):
  if("ASANAR" in w):
    return "CASANARE"
  elif("NTICO" in w and "OFFSHORE" not in w):
    return "ATLANTICO"
  if("BOL" in w):
    return "BOLIVAR"
  if("BOYA" in w):
    return "BOYACA"
  if("COR" in w):
    return "CORDOBA"
  if("DOBA" in w):
    return "CORDOBA"
  else:
    return w
